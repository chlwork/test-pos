        // 導覽切換

        function switchTab(tab) {

            document.querySelectorAll('.view-section').forEach(s => s.classList.add('hidden'));

            document.getElementById(`view-${tab}`).classList.remove('hidden');

            

            document.querySelectorAll('nav button').forEach(b => b.classList.remove('bg-indigo-600', 'shadow'));

            document.getElementById(`tab-${tab}`).classList.add('bg-indigo-600', 'shadow');

            

            const cartEl = document.getElementById('view-cart');

            if (tab === 'pos') {

                cartEl.classList.remove('hidden');

                cartEl.classList.add('flex');

            } else {

                cartEl.classList.remove('flex');

                cartEl.classList.add('hidden');

            }

            

            if(tab === 'history') {

                toggleHistorySubView(false);

                renderHistory();

            }

            if(tab === 'settings') renderSettings();

            lucide.createIcons();

        }


        // 營收統計視窗切換

        function toggleHistorySubView(showDetail) {

            const main = document.getElementById('history-main');

            if (showDetail) {

                main.classList.add('hidden');

                detail.classList.remove('hidden');

            } else {

                main.classList.remove('hidden');

                detail.classList.add('hidden');

            }

            lucide.createIcons();

        }


        // 渲染結帳頁面商品

        function renderProducts() {

            const container = document.getElementById('product-list-container');

            container.innerHTML = '';

            

            const groups = products.reduce((acc, p) => {

                const cat = p.category || '未分類';

                if (!acc[cat]) acc[cat] = [];

                acc[cat].push(p);

                return acc;

            }, {});


            Object.keys(groups).sort().forEach(group => {

                const section = document.createElement('div');

                section.className = 'space-y-4';

                section.innerHTML = `

                    <div class="flex items-center gap-3 px-1">

                        <div class="h-4 w-1 bg-indigo-500 rounded-full"></div>

                        <h3 class="text-sm font-black text-slate-400 uppercase tracking-widest">${group}</h3>

                    </div>

                    <div class="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-3">

                        ${groups[group].map(p => {

                            const inCart = cart.some(item => item.id === p.id);

                            const outOfStock = p.stock <= 0;

                            return `

                            <button onclick="${outOfStock ? '' : `toggleProductSelection(${p.id})`}" 

                                class="active-scale text-left p-4 rounded-3xl border-2 transition-all flex flex-col justify-between h-32 relative overflow-hidden

                                ${inCart ? 'bg-indigo-50 border-indigo-500 ring-2 ring-indigo-200' : 'bg-white border-slate-100 hover:border-indigo-200 shadow-sm'}

                                ${outOfStock ? 'opacity-40 grayscale cursor-not-allowed border-slate-200 bg-slate-100' : ''}">

                                

                                <div class="font-bold text-slate-700 leading-tight">${p.name}</div>

                                

                                <div class="flex items-end justify-between">

                                    <div class="font-mono font-black text-xl ${inCart ? 'text-indigo-600' : 'text-slate-800'}">$${p.price}</div>

                                    <div class="text-[10px] font-bold ${p.stock < 10 ? 'text-rose-500' : 'text-slate-300'}">存貨 ${p.stock}</div>

                                </div>


                                ${inCart ? '<div class="absolute top-2 right-2 text-indigo-500"><i data-lucide="check-circle-2" class="w-5 h-5"></i></div>' : ''}

                                ${outOfStock ? '<div class="absolute inset-0 flex items-center justify-center bg-slate-100/50 font-black text-slate-400 text-xs rotate-[-15deg] pointer-events-none">SOLD OUT</div>' : ''}

                            </button>`;

                        }).join('')}

                    </div>`;

                container.appendChild(section);

            });

            lucide.createIcons();

        }


        // 結帳功能: 選擇商品

        function toggleProductSelection(pid) {

            const prod = products.find(p => p.id === pid);

            const index = cart.findIndex(i => i.id === pid);

            if (index > -1) {

                cart.splice(index, 1);

            } else {

                cart.push({ ...prod, qty: 1 });

            }

            if (window.navigator.vibrate) window.navigator.vibrate(10);

            renderCart();

            updateCartBadge();

            renderProducts(); 

        }


        // 渲染購物車內容

        function renderCart() {

            const container = document.getElementById('cart-items');

            if (cart.length === 0) {

                container.innerHTML = `

                    <div class="flex flex-col items-center justify-center py-20 text-slate-300">

                        <i data-lucide="shopping-basket" class="w-12 h-12 mb-2 opacity-20"></i>

                        <p class="text-sm font-bold">尚未選擇商品</p>

                    </div>`;

                document.getElementById('total-amount').innerText = '$ 0';

                document.getElementById('btn-checkout').disabled = true;

                return;

            }


            let total = 0;

            container.innerHTML = cart.map(item => {

                total += item.price * item.qty;

                return `

                    <div class="flex items-center gap-3 bg-slate-50 p-3 rounded-2xl border border-slate-100">

                        <div class="flex-1 min-w-0">

                            <div class="font-bold text-sm text-slate-700 truncate">${item.name}</div>

                            <div class="text-[10px] font-mono text-slate-400">$${item.price}</div>

                        </div>

                        <div class="flex items-center bg-white rounded-xl border border-slate-200 overflow-hidden shadow-sm">

                            <button onclick="updateQty(${item.id}, -1)" class="w-8 h-8 font-black text-slate-400 active:bg-slate-100">-</button>

                            <span class="w-6 text-center text-xs font-black text-indigo-600">${item.qty}</span>

                            <button onclick="updateQty(${item.id}, 1)" class="w-8 h-8 font-black text-slate-400 active:bg-slate-100">+</button>

                        </div>

                        <button onclick="removeFromCart(${item.id})" class="text-slate-300 hover:text-rose-500 p-1"><i data-lucide="trash-2" class="w-4 h-4"></i></button>

                    </div>`;

            }).join('');

            

            document.getElementById('total-amount').innerText = `$ ${total}`;

            document.getElementById('btn-checkout').disabled = false;

            calculateChange();

            lucide.createIcons();

        }


        function updateQty(id, delta) {

            const item = cart.find(i => i.id === id);

            if (item) {

                item.qty += delta;

                if (item.qty <= 0) cart = cart.filter(i => i.id !== id);

            }

            renderCart(); updateCartBadge(); renderProducts(); 

        }


        function removeFromCart(id) {

            cart = cart.filter(i => i.id !== id);

            renderCart(); updateCartBadge(); renderProducts();

        }


        function calculateChange() {

            const total = parseInt(document.getElementById('total-amount').innerText.replace('$ ', '')) || 0;

            const received = parseInt(document.getElementById('cash-received').value) || 0;

            const changeBox = document.getElementById('change-amount');

            

            if (received >= total && total > 0) {

                changeBox.innerText = `$ ${received - total}`;

                changeBox.classList.remove('text-slate-400', 'text-rose-500');

                changeBox.classList.add('text-emerald-500');

            } else if (received > 0 && received < total) {

                changeBox.innerText = `-$ ${total - received}`;

                changeBox.classList.remove('text-slate-400', 'text-emerald-500');

                changeBox.classList.add('text-rose-500');

            } else {

                changeBox.innerText = `$ 0`;

                changeBox.classList.remove('text-emerald-500', 'text-rose-500');

                changeBox.classList.add('text-slate-400');

            }

        }


        // 完成結帳

        function checkout() {

            const total = parseInt(document.getElementById('total-amount').innerText.replace('$ ', '')) || 0;

            if (total === 0) return;

            const received = parseInt(document.getElementById('cash-received').value) || 0;

            if (received < total) { alert('實收金額不足！'); return; }


            // 更新原始庫存

            cart.forEach(item => {

                const prod = products.find(p => p.id === item.id);

                if (prod) prod.stock -= item.qty;

            });


            // 存入紀錄

            const record = {

                id: Date.now(),

                time: new Date().toLocaleTimeString('zh-TW', { hour12: false, hour: '2-digit', minute: '2-digit' }),

                date: new Date().toLocaleDateString('zh-TW'),

                items: [...cart],

                total: total

            };

            salesHistory.unshift(record);

            

            // 寫入本地存儲

            localStorage.setItem('pos_products', JSON.stringify(products));

            localStorage.setItem('pos_history', JSON.stringify(salesHistory));

            

            alert(`結帳成功！\n總計: $${total}\n找零: $${received - total}`);

            

            clearCart();

            renderProducts();

            if (document.getElementById('view-cart').classList.contains('expanded')) toggleCart();

        }


        // 渲染營收紀錄

        function renderHistory() {

            const list = document.getElementById('history-list');

            const statsContainer = document.getElementById('stats-summary');

            let totalRev = 0;

            const stats = {};

            

            // 處理最近紀錄與統計

            salesHistory.forEach(h => {

                totalRev += h.total;

                h.items.forEach(item => {

                    if (!stats[item.name]) stats[item.name] = { qty: 0, revenue: 0 };

                    stats[item.name].qty += item.qty;

                    stats[item.name].revenue += (item.qty * item.price);

                });

            });


            // 渲染統計明細

            const sortedStats = Object.entries(stats).sort((a, b) => b[1].revenue - a[1].revenue);

            statsContainer.innerHTML = sortedStats.length > 0 ? sortedStats.map(([name, data]) => `

                <div class="p-5 flex justify-between items-center bg-white">

                    <div>

                        <div class="font-bold text-slate-700">${name}</div>

                        <div class="text-[10px] font-bold text-slate-400 mt-1 uppercase tracking-tighter">累計銷量 ${data.qty}</div>

                    </div>

                    <div class="text-right">

                        <div class="font-mono font-black text-indigo-600">$${data.revenue}</div>

                    </div>

                </div>`).join('') : '<div class="p-20 text-center text-slate-300 font-bold">尚無統計資料</div>';


            // 渲染最近交易

            list.innerHTML = salesHistory.length > 0 ? salesHistory.map(h => `

                <div class="p-4 flex justify-between items-center">

                    <div class="min-w-0 flex-1">

                        <div class="text-[10px] font-bold text-slate-300">${h.time}</div>

                        <div class="text-sm font-bold text-slate-600 truncate">${h.items.map(i => `${i.name}x${i.qty}`).join(', ')}</div>

                    </div>

                    <div class="font-mono font-black text-slate-800 ml-4">$${h.total}</div>

                </div>`).join('') : '<div class="p-10 text-center text-slate-300 text-sm font-medium">尚未有成交紀錄</div>';


            document.getElementById('history-total-revenue').innerText = `$ ${totalRev}`;

            document.getElementById('history-count').innerText = `共計 ${salesHistory.length} 筆成交`;

            lucide.createIcons();

        }

        function openProductModal(id = null) {

            const modal = document.getElementById('product-modal');

            const title = document.getElementById('modal-title');

                document.getElementById('edit-name').value = '';

                document.getElementById('edit-price').value = '';

                document.getElementById('edit-stock').value = '';

        function saveProduct() {

            const id = document.getElementById('edit-id').value;

            const name = document.getElementById('edit-name').value;

            const price = parseInt(document.getElementById('edit-price').value);

            const stock = parseInt(document.getElementById('edit-stock').value);

            const category = document.getElementById('edit-category').value;


            if (!name || isNaN(price)) { alert('請完整填寫名稱與價格！'); return; }


            if (id) {

                const idx = products.findIndex(p => p.id == id);

                products[idx] = { id: parseInt(id), name, price, stock, category };

            } else {

                products.push({ id: Date.now(), name, price, stock, category });

            }


            localStorage.setItem('pos_products', JSON.stringify(products));

            renderProducts();

            renderSettings();

            closeProductModal();

        }


        function deleteProduct(id) {

            if (confirm('確定要刪除此商品嗎？這將無法復原。')) {

                products = products.filter(p => p.id !== id);

                localStorage.setItem('pos_products', JSON.stringify(products));

                renderProducts();

                renderSettings();

            }

        }
